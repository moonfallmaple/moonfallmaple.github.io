Fetch Data (https://davidwalsh.name/fetch)

Fetch Request

fetch(`https://api.unsplash.com/search/photos?page=1&query=${searchedForText}`, {
    method: 'GET'
});

Handle Response (The response object doesn’t have any data inside, to get the data need to get the “body” of the response)

fetch(`https://api.unsplash.com/search/photos?page=1&query=${searchedForText}`, {
    headers: {
        Authorization: 'Client-ID abc123'
    }
}).then(function(response) {
    return response.json();
})

The response Object (convert the response to json and pass the data to addImage function)

fetch(`https://api.unsplash.com/search/photos?page=1&query=${searchedForText}`, {
    headers: {
        Authorization: 'Client-ID abc123'
    }
}).then(function(response) {
    return response.json();
}).then(addImage);

function addImage(data) {
    let htmlContent = '';
    const firstImage = data.results[0];

    if (firstImage) {
        htmlContent = `<figure>
            <img src="${firstImage.urls.small}" alt="${searchedForText}">
            <figcaption>${searchedForText} by ${firstImage.user.name}</figcaption>
        </figure>`;
    } else {
        htmlContent = 'Unfortunately, no image was returned for your search.'
    }

    responseContainer.insertAdjacentHTML('afterbegin', htmlContent);
}

fetch('https://davidwalsh.name/flowers.jpg')
    .then(function(response) {
      return response.blob();
    }) 
    .then(function(imageBlob) {
      document.querySelector('img').src = URL.createObjectURL(imageBlob);
